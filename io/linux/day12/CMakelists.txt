set(CMAKE_CXX_STANADRD_REQUIRED ON)

		VERSION 0.1
		DESCRIPTION "pine"

		)

# common case of running it from the root directory
file(TO_CMAKE_PATH "${PROJECT_BINARY_DIR}/CMakeLists.txt"
